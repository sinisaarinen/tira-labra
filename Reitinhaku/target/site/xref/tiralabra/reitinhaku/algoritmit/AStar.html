6   package tiralabra.reitinhaku.algoritmit;
7   
8   import tiralabra.reitinhaku.tietorakenteet.Keko;
9   import tiralabra.reitinhaku.tietorakenteet.Solmu;
10  
11  /**
12   *
13   * Luokka vastaa A*-algoritmin toteutuksesta.
14   */
15  public class AStar {
16      
17      private Keko minimiKeko;
18      private char[][] kartta;
19      
20      public AStar(char[][] kartta) {
21          this.kartta = kartta;
22          this.minimiKeko = new Keko(kartta.length * kartta[0].length);
23      }
24      /**
25       * Metodi saa parametreikseen alku- ja loppusolmut. Se asettaa alkusolmulle
26       * alkuetäisyyden sekä arvioi kartan kullekin solmulle etäisyydet loppuun.
27       * 
28       * @param alku solmu, josta reitti alkaa
29       * @param loppu solmu, johon reitti päättyy
30       */
31      public void alusta(Solmu alku, Solmu loppu) {
32          alku.setLyhinEtaisyysAlusta(0);
33          for (int i = 0; i < kartta.length; i++) {
34              for (int j = 0; j < kartta[0].length; j++) {
35                  Solmu solmu = new Solmu(i, j);
36                  arvioiEtaisyysLoppuun(solmu, loppu);
37              }
38          }
39      }
40      /**
41       * Metodi arvioi ja asettaa etäisyydet nykyisestä solmusta loppuun.
42       * 
43       * @param solmu nykyinen solmu
44       * @param loppu solmu, johon reitti päättyy
45       */
46      public void arvioiEtaisyysLoppuun(Solmu solmu, Solmu loppu) {
47          int etaisyysLoppuun = 
48                  (Math.abs(loppu.getX() - solmu.getX())
49                  + Math.abs(loppu.getY() - solmu.getY()));
50          solmu.setEtaisyysArvioLoppuun(etaisyysLoppuun);
51      }
52  }

6   package tiralabra.reitinhaku.algoritmit;
7   
8   import tiralabra.reitinhaku.tietorakenteet.Keko;
9   import tiralabra.reitinhaku.tietorakenteet.Solmu;
10  
11  /**
12   *
13   * Luokka vastaa Dijkstran algoritmin toteutuksesta.
14   */
15  public class Dijkstra {
16      
17      private Keko minimiKeko;
18      private char[][] kartta;
19      private int lyhinReitti;
20      private int[][] etaisyys;
21      private boolean[][] kasitelty;
22      
23      public Dijkstra(char[][] kartta) {
24          this.kartta = kartta;
25          this.minimiKeko = new Keko(kartta.length * kartta[0].length);
26          this.lyhinReitti = 0;
27          this.etaisyys = new int[kartta.length][kartta[0].length];
28          this.kasitelty = new boolean[kartta.length][kartta[0].length];
29          for (int i = 0; i < kartta.length; i++) {
30              for (int j = 0; j < kartta[0].length; j++) {
31                  etaisyys[i][j] = Integer.MAX_VALUE;
32                  kasitelty[i][j] = false;
33              }
34          }
35      }
36      /**
37       * Metodi saa parametreikseen reitin alku- ja loppusolmut ja laskee niiden
38       * välisen lyhyimmän reitin. Alkusolmu lisätään kekoon, jonka jälkeen kekoa
39       * käsitellään kunnes se on tyhjä. Jos käsiteltävä solmu on sama kuin loppu-
40       * solmu, kutsutaan metodia reitinPituus. Jos alkusolmusta ei ole mahdollista
41       * päästä loppusolmuun, metodi palauttaa -1.
42       * 
43       * @param alku solmu, josta reitti alkaa
44       * @param solmu, johon reitti päättyy
45       * @return reitin pituus
46       */
47      public int laskeReitinPituus(Solmu alku, Solmu loppu) {
48          this.minimiKeko.lisaaSolmu(alku);
49  
50          while (!minimiKeko.isEmpty()) {
51              Solmu kasiteltava = minimiKeko.poistaPienin();
52              kasitelty[kasiteltava.getX()][kasiteltava.getY()] = true;
53  
54              if (kasiteltava.equals(loppu)) {
55                  return reitinPituus(kasiteltava);
56              }
57              etsi(kasiteltava, loppu);
58          }
59          return -1;
60      }
61      /**
62       * Metodi saa parametrikseen solmun, johon reitti päättyy. Se laskee reitin
63       * pituuden etsimällä aina solmun vanhemman, kunnes vanhempia ei enää ole.
64       * Tällöin reitin pituus on saatu laskettua.
65       * 
66       * @param solmu solmu, johon reitti päättyy
67       * @return reitin pituus
68       */
69      public int reitinPituus(Solmu solmu) {
70          if (solmu.getVanhempi() != null) {
71              lyhinReitti++;
72              reitinPituus(solmu.getVanhempi());
73          }
74          return lyhinReitti;
75      }
76      /**
77       * Metodi saa parametreikseen käsiteltävän solmun ja loppusolmun. Se kutsuu
78       * metodia tutkiEtaisyyksia kaikkien solmun naapureiden osalta tarkistaakseen,
79       * voidaanko naapurisolmujen etäisyyksiä päivittää.
80       * 
81       * @param kasiteltava käsiteltävä solmu
82       * @param loppu loppusolmu
83       */
84      public void etsi(Solmu kasiteltava, Solmu loppu) {
85          //askel ylös
86          tutkiEtaisyyksia(kasiteltava.getX() - 1, kasiteltava.getY(), kasiteltava, loppu);
87          //askel alas
88          tutkiEtaisyyksia(kasiteltava.getX() + 1, kasiteltava.getY(), kasiteltava, loppu);
89          //askel vasemmalle
90          tutkiEtaisyyksia(kasiteltava.getX(), kasiteltava.getY() - 1, kasiteltava, loppu);
91          //askel oikealle
92          tutkiEtaisyyksia(kasiteltava.getX(), kasiteltava.getY() + 1, kasiteltava, loppu);  
93      }
94      /**
95       * Metodi tarkistaa ensin, onko annettu solmu kartalla ja onko siihen mahdollista
96       * edetä. Tämän jälkeen solmulle lasketaan etäisyys alusta ja se lisätään kekoon.
97       * 
98       * @param y solmun y-koordinaatti
99       * @param x solmun x-koordinaatti
100      * @param edeltaja edeltäjäsolmu
101      * @param loppu loppusolmu
102      */
103     public void tutkiEtaisyyksia(int x, int y, Solmu edeltaja, Solmu loppu) {
104         if ((x >= 0 && x < kartta.length) && (y >= 0 && y < kartta[0].length) && kartta[x][y] == '.' && kasitelty[x][y] == false) {
105             if (edeltaja.getLyhinEtaisyysAlusta() + 1 < etaisyys[x][y]) {
106                 int etaisyysAlusta = edeltaja.getLyhinEtaisyysAlusta() + 1;
107                 etaisyys[x][y] = etaisyysAlusta;
108                 Solmu solmu = new Solmu(x, y, etaisyysAlusta, edeltaja);                
109                 minimiKeko.lisaaSolmu(solmu);
110             }
111         }
112     }
113     
114     public Keko getKeko() {
115         return this.minimiKeko;
116     }
117 }

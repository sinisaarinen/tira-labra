6   package tiralabra.reitinhaku.main;
7   
8   import java.io.File;
9   import java.io.FileNotFoundException;
10  import tiralabra.reitinhaku.algoritmit.AStar;
11  import tiralabra.reitinhaku.algoritmit.Dijkstra;
12  import tiralabra.reitinhaku.kartat.Kartanlukija;
13  import tiralabra.reitinhaku.tietorakenteet.Solmu;
14  import tiralabra.reitinhaku.ui.Kayttoliittyma;
15  /**
16   *
17   * @author saasini
18   */
19  public class Main {
20  
21      public static void main(String[] args) throws FileNotFoundException, Exception {
22          File kartta = new File("./kartat/London_2_256.map");
23          Kartanlukija kartanlukija = new Kartanlukija(kartta);
24          char[][] valittuKartta = kartanlukija.muutaMatriisiksi();
25          
26          Solmu alku = new Solmu(5, 2);
27          Solmu loppu = new Solmu(200, 200);
28          Dijkstra dijkstra = new Dijkstra(valittuKartta);
29          AStar astar = new AStar(valittuKartta);
30          
31          long aloitusAika = System.currentTimeMillis();
32          int lyhinReitti = dijkstra.laskeReitinPituus(alku, loppu);
33          long lopetusAika = System.currentTimeMillis();
34          
35          long aloitusAika2 = System.currentTimeMillis();
36          int lyhinReitti2 = astar.laskeReitti(alku, loppu);
37          long lopetusAika2 = System.currentTimeMillis();
38  
39          System.out.println("Lyhin reitti Dijkstran algoritmilla oli pituudeltaan " + lyhinReitti);
40          System.out.println("Lyhin reitti A*-algoritmin etsimänä oli pituudeltaan " + lyhinReitti2);
41          System.out.println();
42          System.out.println("Suoritusaika Dijkstran algoritmille oli " + (lopetusAika - aloitusAika) + "ms.");
43          System.out.println("Suoritusaika A*-algoritmille oli " + (lopetusAika2 - aloitusAika2) + "ms.");
44          
45          Kayttoliittyma.main(args);
46      }
47  }

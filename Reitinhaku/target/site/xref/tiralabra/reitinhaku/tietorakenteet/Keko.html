6   package tiralabra.reitinhaku.tietorakenteet;
7   
8   /**
9    *
10   * Luokka vastaa minimikeko-tietorakenteesta. Keko on tallennettu taulukkona.
11   */
12  public class Keko {
13      
14      private int koko;
15      private Solmu[] solmut;
16      
17      public Keko(int koko) {
18          this.koko = -1;
19          this.solmut = new Solmu[koko];
20      }
21      
22      public Solmu getPienin() {
23          return solmut[0];
24      }
25      /**
26       * Metodi poistaa keon pienimmän solmun ja palauttaa sen. Metodi myös kutsuu
27       * painaAlas-metodia, jotta keko pysyy järjestyksessä.
28       * @return keon pienin solmu
29       */
30      public Solmu poistaPienin() {
31          Solmu arvo = solmut[0];
32          vaihdaSolmut(0, koko);
33          this.koko--;
34          painaAlas(0);
35          return arvo;
36      }
37       /**
38       * Metodi vertailee kahden solmun paikat.
39       * @param sijainti1 ensimmäisen solmun sijainti keossa
40       * @param sijainti2 toisen solmun sijainti keossa
41       */
42      public void vaihdaSolmut(int sijainti1, int sijainti2) {
43          Solmu apu = solmut[sijainti1];
44          solmut[sijainti1] = solmut[sijainti2];
45          solmut[sijainti2] = apu;
46          if (solmut[sijainti1] != null) {
47              solmut[sijainti1].setSijainti(sijainti1);
48          }
49          if (solmut[sijainti2] != null) {
50              solmut[sijainti2].setSijainti(sijainti2);
51          }
52      }
53      /**
54       * Metodi, joka huolehtii siitä, että minimikeon kekovaatimus toteutuu. Metodi
55       * kutsuu vaihdaSolmut-metodia ja itseään rekursiivisesti, kunnes kekovaatimus
56       * täyttyy.
57       * 
58       * @param sijainti solmun sijainti keossa
59       */
60      public void painaAlas(int sijainti) {
61          int vasen = etsiVasenLapsi(sijainti);
62          int oikea = etsiOikeaLapsi(sijainti);
63          int pienin;
64          int apu;
65          
66          if (oikea <= koko) { //solmulla on molemmat lapset
67              if (solmut[vasen].compareTo(solmut[oikea]) == -1 || solmut[oikea] == null) { //valitaan pienempi lapsista
68                  pienin = vasen;
69              } else {
70                  pienin = oikea;
71              }
72              if (solmut[sijainti].compareTo(solmut[pienin]) == 1) {
73                  vaihdaSolmut(sijainti, pienin);
74                  painaAlas(pienin);
75              }
76          } else if (solmut[vasen] != null && vasen == koko && solmut[sijainti].compareTo(solmut[vasen]) == 1) {
77              vaihdaSolmut(sijainti, vasen);
78          }
79      }
80       /**
81       * Metodi laskee solmun vasemman lapsen sijainnin ja palauttaa sen.
82       * 
83       * @param sijainti solmun sijainti keossa
84       * @return vasemman lapsen sijainti
85       */
86      public int etsiVasenLapsi(int sijainti) {
87          return 2 * sijainti;
88      }
89       /**
90       * Metodi laskee solmun oikean lapsen sijainnin ja palauttaa sen.
91       * 
92       * @param sijainti solmun sijainti keossa
93       * @return oikean lapsen sijainti
94       */
95      public int etsiOikeaLapsi(int sijainti) {
96          return 2 * sijainti + 1;
97      }
98  }

6   package tiralabra.reitinhaku.kartat;
7   
8   import java.io.File;
9   import java.io.FileNotFoundException;
10  import java.util.Scanner;
11  
12  /**
13   *
14   * Luokka vastaa .map-tyyppisten karttatiedostojen lukemisesta ja niiden
15   * muuntamisesta matriisimuotoon.
16   */
17  public class Kartanlukija {
18      
19      private File kartta;
20      private Scanner lukija;
21      
22       /**
23       * Konstruktori saa parametrinaan tiedoston ja kutsuu lueKartta-metodia.
24       * @param tiedosto
25       * @throws FileNotFoundException 
26       */
27      public Kartanlukija(File tiedosto) throws FileNotFoundException {
28          this.kartta = tiedosto;        
29          if (!lueKartta()) {
30              throw new FileNotFoundException();
31          }        
32      }
33      /**
34       * Metodi saa parametrinaan tiedoston ja kutsuu lueKartta-metodia.
35       * @param tiedosto
36       * @return true jos lueKartta-metodi palauttaa true, muuten false
37       */
38      public boolean lataaKartta(File tiedosto) {
39          kartta = tiedosto;
40          return lueKartta();
41      }
42      /**
43       * Metodi lukee .map-tiedoston.
44       * @return true jos tiedosto l√∂ytyy, muuten false
45       */
46      public boolean lueKartta() {
47          try {            
48              lukija = new Scanner(kartta);
49              return true;
50          } catch (FileNotFoundException poikkeus) {
51              return false;
52          }
53      }
54       /**
55       * Metodi luo matriisin .map-muotoisesta tiedostosta.
56       * @return matriisi, jos tiedoston luku onnistuu.
57       * @throws Exception 
58       */
59      public char[][] luoMatriisi() throws Exception {
60          String korkeus;
61          String leveys;
62          try {
63              lukija.nextLine();
64              korkeus = lukija.nextLine();
65              leveys = lukija.nextLine();            
66              char[][] matriisi = 
67                  new char[Integer.parseInt(korkeus.substring(7))]
68                          [Integer.parseInt(leveys.substring(6))];
69              lukija.nextLine();                      
70              return matriisi;
71          } catch (Exception poikkeus) {
72              throw new Exception("Valittu tiedosto ei kelpaa.");            
73          }        
74      }
75       /**
76       * Metodi kutsuu luoMatriisi-metodia ja palauttaa luodun matriisin.
77       * @return luotu matriisi
78       * @throws Exception 
79       */
80      public char[][] muutaMatriisiksi() throws Exception {
81          char[][] matriisi = luoMatriisi();
82          for (int i = 0; i < matriisi.length; i++) {
83              matriisi[i] = lukija.nextLine().toCharArray();
84          }
85          return matriisi;
86      }
87  }

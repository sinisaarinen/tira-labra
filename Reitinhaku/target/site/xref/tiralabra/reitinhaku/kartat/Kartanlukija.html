6   package tiralabra.reitinhaku.kartat;
7   
8   import java.io.File;
9   import java.io.FileNotFoundException;
10  import java.util.Scanner;
11  
12  /**
13   *
14   * @author saasini
15   */
16  public class Kartanlukija {
17      private File kartta;
18      private Scanner lukija;
19      
20      public Kartanlukija(File tiedosto) throws FileNotFoundException {
21          this.kartta = tiedosto;        
22          if (!lueKartta()) {
23              throw new FileNotFoundException();
24          }        
25      }
26      
27      public boolean lataaKartta(File tiedosto) {
28          kartta = tiedosto;
29          return lueKartta();
30      }
31      
32      public boolean lueKartta() {
33          try {            
34              lukija = new Scanner(kartta);
35              return true;
36          } catch (FileNotFoundException poikkeus) {
37              return false;
38          }
39      }
40      
41      public char[][] luoMatriisi() throws Exception {
42          String korkeus;
43          String leveys;
44          try {
45              lukija.nextLine();
46              korkeus = lukija.nextLine();
47              leveys = lukija.nextLine();            
48              char[][] matriisi = 
49                  new char[Integer.parseInt(korkeus.substring(7))]
50                          [Integer.parseInt(leveys.substring(6))];
51              lukija.nextLine();                      
52              return matriisi;
53          } catch (Exception poikkeus){
54              throw new Exception("Valittu tiedosto ei kelpaa.");            
55          }        
56      }
57      
58      public char[][] muutaMatriisiksi() throws Exception {
59          char[][] matriisi = luoMatriisi();
60          for (int i = 0; i < matriisi.length; i++) {
61              matriisi[i] = lukija.nextLine().toCharArray();
62          }
63          return matriisi;
64      }
65  }

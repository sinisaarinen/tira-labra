6   package tiralabra.reitinhaku.logiikka;
7   
8   import java.io.File;
9   import java.io.FileNotFoundException;
10  import tiralabra.reitinhaku.kartat.Kartanlukija;
11  /**
12   *
13   * Luokka vastaa sovelluslogiikasta.
14   */
15  public class Logiikka {
16      
17      private String valittuKartta;
18      private File kartta;
19      private Kartanlukija kartanlukija;
20      private char[][] matriisi;
21      private String valittuAlgo;
22      /**
23       * Sovelluslogiikka alustetaan oletuksena Lontoon kartalla ja Dijkstran
24       * algoritmilla, jos käyttäjä ei itse valitse karttaa tai algoritmia.
25       * 
26       * @throws FileNotFoundException
27       * @throws Exception
28       */
29      public Logiikka() throws FileNotFoundException, Exception {
30          this.valittuKartta = "kartta1";
31          this.kartta = new File("./kartat/London_2_256.map");
32          this.kartanlukija = new Kartanlukija(kartta);
33          this.matriisi = kartanlukija.muutaMatriisiksi();
34          this.valittuAlgo = "Dijkstra";
35      }
36      
37      public void setValittuKartta(String kartta) {
38          this.valittuKartta = kartta;
39      }
40      /**
41       * Metodi palauttaa matriisimuotoisen kartan käyttäjän valinnan perusteella.
42       * 
43       * @throws FileNotFoundException
44       * @throws Exception
45       * 
46       * @return matriisimuotoinen kartta
47       */
48      public char[][] getValittuKartta() throws FileNotFoundException, Exception {
49          if (this.valittuKartta.equals("kartta1")) {
50              File kartta1 = new File("./kartat/London_2_256.map");
51              Kartanlukija kartanlukija = new Kartanlukija(kartta1);
52              this.matriisi = kartanlukija.muutaMatriisiksi();
53          } else if (this.valittuKartta.equals("kartta2")) {
54              File kartta2 = new File("./kartat/Milan_2_256.map");
55              Kartanlukija kartanlukija = new Kartanlukija(kartta2);
56              this.matriisi = kartanlukija.muutaMatriisiksi();
57          } else if (this.valittuKartta.equals("kartta3")) {
58              File kartta3 = new File("./kartat/Moscow_1_256.map");
59              Kartanlukija kartanlukija = new Kartanlukija(kartta3);
60              this.matriisi = kartanlukija.muutaMatriisiksi();
61          } else if (this.valittuKartta.equals("kartta4")) {
62              File kartta4 = new File("./kartat/NewYork_1_256.map");
63              Kartanlukija kartanlukija = new Kartanlukija(kartta4);
64              this.matriisi = kartanlukija.muutaMatriisiksi();
65          }
66          return this.matriisi;
67      }
68      
69      public void setValittuAlgo(String algoritmi) {
70          this.valittuAlgo = algoritmi;
71      }
72      
73      public String getValittuAlgo() {
74          return this.valittuAlgo;
75      }
76  }

6   package tiralabra.reitinhaku.logiikka;
7   
8   import java.io.File;
9   import java.io.FileNotFoundException;
10  import tiralabra.reitinhaku.kartat.Kartanlukija;
11  
12  /**
13   *
14   * @author saasini
15   */
16  public class Logiikka {
17      
18      private String valittuKartta;
19      private File kartta;
20      private Kartanlukija kartanlukija;
21      private char[][] matriisi;
22      
23      public Logiikka() throws FileNotFoundException, Exception {
24          this.valittuKartta = "kartta1";
25          this.kartta = new File("./kartat/London_2_256.map");
26          this.kartanlukija = new Kartanlukija(kartta);
27          this.matriisi = kartanlukija.muutaMatriisiksi();
28      }
29      
30      public void setValittuKartta(String kartta) throws FileNotFoundException, Exception {
31          this.valittuKartta = kartta;
32      }
33      
34      public char[][] getValittuKartta() throws FileNotFoundException, Exception {
35          if (this.valittuKartta.equals("kartta1")) {
36              File kartta1 = new File("./kartat/London_2_256.map");
37              Kartanlukija kartanlukija = new Kartanlukija(kartta1);
38              this.matriisi = kartanlukija.muutaMatriisiksi();
39          } else if (this.valittuKartta.equals("kartta2")) {
40              File kartta2 = new File("./kartat/Milan_2_256.map");
41              Kartanlukija kartanlukija = new Kartanlukija(kartta2);
42              this.matriisi = kartanlukija.muutaMatriisiksi();
43          } else if (this.valittuKartta.equals("kartta3")) {
44              File kartta3 = new File("./kartat/Moscow_1_256.map");
45              Kartanlukija kartanlukija = new Kartanlukija(kartta3);
46              this.matriisi = kartanlukija.muutaMatriisiksi();
47          } else if (this.valittuKartta.equals("kartta4")) {
48              File kartta4 = new File("./kartat/NewYork_1_256.map");
49              Kartanlukija kartanlukija = new Kartanlukija(kartta4);
50              this.matriisi = kartanlukija.muutaMatriisiksi();
51          }
52          return this.matriisi;
53      }
54  }
